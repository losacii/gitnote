function loban() {
  # lolcat banner: display a stylish message!
  echo -e '\n'
  figlet -w 166 -f Bloody $1 | lolcat
  #figlet -w 160 -c -f Bloody $1 | lolcat
}

function xloban() {
  echo;
  figlet -w 166 -f 3d $1 | lolcat

}

function pskill() {
  ps aux | fzy | awk '{ print $2 }' | xargs kill
}

    PS1='\n\[\033[01;31m\]->\[\033[00m\] '
}
